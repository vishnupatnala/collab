function sendMessage() { const input = document.getElementById('userInput'); const text = input.value.trim(); if (!text) return; addMessage(text, 'user'); input.value = ''; // Show typing... const typingMsg = document.createElement('div'); typingMsg.classList.add('message', 'bot'); typingMsg.textContent = 'Bot is typing...'; chat.appendChild(typingMsg); chat.scrollTop = chat.scrollHeight; // Simulate delay then reply setTimeout(() => { chat.removeChild(typingMsg); const reply = getBotReply(text); addMessage(reply, 'bot'); }, 1000); } function getBotReply(message) { message = message.toLowerCase(); const responses = [ { keyword: ['hello', 'hi'], reply: "Hi there! How can I help you?" }, { keyword: ['help'], reply: "Sure, I'm here to help! Ask me anything." }, { keyword: ['time'], reply: "Current time is: " + new Date().toLocaleTimeString() }, { keyword: ['bye', 'goodbye'], reply: "Goodbye! Have a nice day!" } ]; for (let res of responses) { if (res.keyword.some(k => message.includes(k))) { return res.reply; } } return "Sorry, I didn't understand that. Type 'help' for assistance."; } document.getElementById('userInput').focus(); document.getElementById('userInput').addEventListener('keydown', function(e) { if (e.key === 'Enter') sendMessage(); });
